
SGRP - Shader Group
	SFAM - Shader Family
		int - id
		string family name
		string surface properties file
		bit - R
		bit - G
		bit - B
		float - ? always seems to be 0x00000040 64
		float - feather clamp
		int - child count
		shader child {
			string family data - name
			float shader data - weight
		}
FGRP - Flora Group
	FFAM - Flora Family
		int - id
		string - name
		family data {
			bit - R
			bit - G
			bit - B
			float - density
		}
		bool/int - floats on water
		int - child count
		flora child {
			family data {
				string name
			}
			child data {
				float weight
				bool/int align to terrain
			}
			swaying {
				float displacement? 
				float period?
				bool/int sway flora?
			}
			scaling {
				bool/int scale flora
				float min scale
				float max scale
			}
		}
